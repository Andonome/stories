\mapentry[twoGuards]{Two \Glsfmtplural{sunGuard} Arrive}
with \pgls{torch} and swords.
Your only choice, is to fight.
Make \pgls{combat} roll: $2D6$ plus \roll{Dexterity}{Brawl}.

\begin{selectPath}
  \willYe{Roll under 10}%
  {and \pgls{sunGuard} cuts you, dealing \dmg{3}~Damage.
  You can spend \glspl{fp} to reduce the Damage.}%
  {}
  \willYe{Roll 10}%
  {and you dodge.}%
  {}
  \willYe{Roll over 10}%
  {and you strike at their chainmail armour.
  This deals \dmg{0}~Damage plus your Strength Bonus, but their armour reduces the Damage by~4.}%
  {}
  \willYe{Roll 13 or more}%
  {and you strike their face, dealing \dmg{0}, plus your Strength Bonus.  If you have a dagger, add +2 to your Damage.}%
  {}
  \willYe{Once \pgls{sunGuard} receives 3~Damage}%
  {he backs off.}%
  {}
  \willYe{While one remains standing}%
  {roll to attack again.}%
  {}
  \willYe{If both of the \glspl{sunGuard} back off}%
  {you push through and run down the hall, in part \vref{darkHall}.}%
  {}
\end{selectPath}
